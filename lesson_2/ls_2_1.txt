Префиксная форма ++variable увеличивает variable на 1 и возвращает новое значение 2.
Постфиксная форма variable++ тоже увеличивает variable на 1, но возвращает старое значение 1.
Например:

var a = 1, b, c;
b = ++a; alert(b); // тут сначала а увеличивается, потом результат присваивается к b
с = a++; alert(с); // тут сначала результат присваивается к с, потом а увеличивается

И в примере из дзшки:

var a = 1, b = 1, c, d;
c = ++a; alert(c);           // Ответ: 2. Тут а принимает значение 1+1=2 и после этого присваивается к с
d = b++; alert(d);           // Ответ: 1. Тут b присваивается к d и после этого принимает значение 1+1=2
c = (2+ ++a); alert(c);      // Ответ: 5. Изначально а = 2, к нему добавляется 1, так как ++ префиксная. Результат 3 суммируется с 2 в начале выражения
d = (2+ b++); alert(d);      // Ответ: 4. Изначально b = 2, к нему добавляется 1, ПОСЛЕ того как значение b суммируется с 2 в начале выражения
alert(a);                    // Ответ: 3 ++а выполняется в коде 2 раза, то есть 1+1+1
alert(b);                    // Ответ: 3 b++ выполняется в коде 2 раза, то есть 1+1+1
